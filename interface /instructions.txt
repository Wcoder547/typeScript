ğŸ“Œ 5. Practical Examples

// Interface extension (inheritance)
interface Animal {
    name: string;
}
interface Dog extends Animal {
    bark(): void;
}
const myDog: Dog = {
    name: "Tommy",
    bark: () => console.log("Woof")
};

// Type intersection
type AnimalType = {
    name: string;
};
type DogType = AnimalType & {
    bark(): void;
};
const myDog2: DogType = {
    name: "Rocky",
    bark: () => console.log("Woof!")
};

// Type with union and primitives
type ID = string | number;
type Point = { x: number; y: number } | [number, number];

// Interface merging
interface Box {
    height: number;
}
interface Box {
    width: number;
}
const myBox: Box = { height: 10, width: 20 };

// Type - cannot merge
type BoxType = {
    height: number;
};
